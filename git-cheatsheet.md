# git cheatssheet

# общие консольные команды

1. mkdir - создает папку
2. touch - создает файл
3. dotnet new console -o Program --use-program-main - создание консольного приложения на c#

# начало работы

1. git version - убеждаемся, что гит установлен корректно, выводя его версию
2. git config --global user.name "ник"
   git config --global user.email "почта" - указываем данные для подписи коммитов
3. git init -b main - инициализация локального репозитория и ветки main

# команды для добавления файлов в репозиторий

1. git status - проверка состояния репозитория
2. git add . - добавление в локальный репозиторий всех файлов
3. git commit -m "текст коммита" - создание коммита
4. git log - показ истории коммитов
5. git push -u origin main - пушим изменения на сервер

# ветки

1. git checkout -b new-feature - создание новой ветки с переключением
2. git branch - просмотр всех веток
3. git branch название-ветки - создание ветки
4. git checkout - переключиться на ветку
5. git switch - переключиться на ветку
6. git switch -c - создать и переключиться
7. git branch -b - удалить ветку
8. git pull - обновление ветки с сервера
9. git fetch --prune - загружаем с сервера информацию об удаленных ветках
